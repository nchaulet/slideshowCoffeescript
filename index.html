Objet et d'heritage


	# classe qui représente un livre
	class Livre
		constructor: (@titre, @datePublication)

		lire : () ->
			console.log @titre+' lu '

		@staticMethode : () ->
			console.log 'méthode statique!'
					

	# classe qui représente une bande dessinée
	class BD extends Livre
		constructor: (@titre, @datePublication)

		lire : () ->
			console.log @titre+' regardé '

		@staticMethode : () ->
			console.log 'méthode statique!'
